

include_directories("${MAIN_PROJECT_DIR}/source")
include_directories("${EXTERNALS_DIR}/engine/include")
include_directories("${EXTERNALS_DIR}/glm")

if (DEBUG_BUILD)
	set_target_properties(Tutorial PROPERTIES VS_DEBUGGER_WORKING_DIRECTORY "${EXECUTABLE_OUTPUT_PATH}/Debug")
else (DEBUG_BUILD)
	set_target_properties(Tutorial PROPERTIES VS_DEBUGGER_WORKING_DIRECTORY "${EXECUTABLE_OUTPUT_PATH}/Release")
endif (DEBUG_BUILD)

if (DEBUG_BUILD)
  if (USE_64_BIT)
    target_link_libraries(Tutorial "${EXTERNALS_DIR}/engine/lib/x64/Engine_d.lib")
  else (USE_64_BIT)
    target_link_libraries(Tutorial "${EXTERNALS_DIR}/engine/lib/x32/Engine_d.lib")
  endif (USE_64_BIT)
else (DEBUG_BUILD)
  if (USE_64_BIT)
    target_link_libraries(Tutorial "${EXTERNALS_DIR}/engine/lib/x64/Engine.lib")
  else (USE_64_BIT)
    target_link_libraries(Tutorial "${EXTERNALS_DIR}/engine/lib/x32/Engine.lib")
  endif (USE_64_BIT)
endif (DEBUG_BUILD)

if (USE_GLFW)
  if (USE_64_BIT)
    find_library(glfw_library glfw3.lib "${EXTERNALS_DIR}/glfw-Win64/lib-vc2015")
  else (USE_64_BIT)
    find_library(glfw_library glfw3.lib "${EXTERNALS_DIR}/glfw-Win32/lib-vc2015")
  endif (USE_64_BIT)
  
  include_directories("${EXTERNALS_DIR}/glfw/include")
  target_link_libraries(Tutorial ${glfw_library})
  if (WIN32)
    find_library(legacy_stdio_library legacy_stdio_definitions.lib "${VISUAL_STUDIO_DIR}")
    target_link_libraries(Tutorial ${legacy_stdio_library})
  endif (WIN32)
endif(USE_GLFW)

if (USE_SDL)
  #find_library(sdl_library sdl2.lib "${EXTERNALS_DIR}/sdl/lib/x64")
  include_directories("${EXTERNALS_DIR}/sdl/include")
  add_library(sdl_lib SHARED IMPORTED)
  if (USE_64_BIT)
    set_property(TARGET sdl_lib PROPERTY IMPORTED_LOCATION "${EXTERNALS_DIR}/sdl/lib/x64/SDL2.dll")
    set_property(TARGET sdl_lib PROPERTY IMPORTED_IMPLIB "${EXTERNALS_DIR}/sdl/lib/x64/SDL2.lib")
  else (USE_64_BIT)
    set_property(TARGET sdl_lib PROPERTY IMPORTED_LOCATION "${EXTERNALS_DIR}/sdl/lib/x86/SDL2.dll")
    set_property(TARGET sdl_lib PROPERTY IMPORTED_IMPLIB "${EXTERNALS_DIR}/sdl/lib/x86/SDL2.lib")
  endif (USE_64_BIT)
  target_link_libraries(Tutorial sdl_lib)
endif(USE_SDL)

if (USE_WINAPI)
  target_link_libraries(Tutorial d3d11.lib)
  target_link_libraries(Tutorial d3dcompiler.lib)
endif (USE_WINAPI)
